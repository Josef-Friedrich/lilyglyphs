% Try to get by number
% created by genGlyphCommands.py on 2012-11-09
\newcommand*{\byNumberBase}[1][]{%
	\setkeys{lilyDesignOptions}{scale=1,raise=0}%
	\lilyPrint[#1]{\lilyGetGlyphByNumber{33}}%
}
\newcommand*{\byNumber}[1][]{\byNumberBase[#1] }
\WithSuffix\newcommand\byNumber*[1][]{\byNumberBase[#1]}

% downward fermata
% created by genGlyphCommands.py on 2012-11-09
\newcommand*{\fermataDownBase}[1][]{%
	\setkeys{lilyDesignOptions}{scale=1,raise=0}%
	\lilyPrint[#1]{\lilyGetGlyph{scripts.dfermata}}%
}
\newcommand*{\fermataDown}[1][]{\fermataDownBase[#1] }
\WithSuffix\newcommand\fermataDown*[1][]{\fermataDownBase[#1]}

% Rinforzando, to be kerned
% created by genGlyphCommands.py on 2012-11-09
\newcommand{\rinforzandoBase}[1][]{%
	\mbox{%
		\lilyDynamics[#1]{rfz}%
	}%
}
\newcommand*{\rinforzando}[1][]{\rinforzandoBase[#1] }
\WithSuffix\newcommand\rinforzando*[1][]{\rinforzandoBase[#1]}

% short (triangular) fermata
% created by genGlyphCommands.py on 2012-11-09
\newcommand*{\shortFermataBase}[1][]{%
	\setkeys{lilyDesignOptions}{scale=1,raise=0}%
	\lilyPrint[#1]{\lilyGetGlyph{scripts.ushortfermata}}%
}
\newcommand*{\shortFermata}[1][]{\shortFermataBase[#1] }
\WithSuffix\newcommand\shortFermata*[1][]{\shortFermataBase[#1]}

% created by genGlyphCommands.py on 2012-11-09
\newcommand{\textTwoBase}[1][]{%
	\mbox{%
		\lilyText[#1]{5 - 4 ff}%
	}%
}
\newcommand*{\textTwo}[1][]{\textTwoBase[#1] }
\WithSuffix\newcommand\textTwo*[1][]{\textTwoBase[#1]}
