% dotted quaver with upward stem
% created by build-image-commands.py on 2012-12-03

\newcommand*{\quaverDottedBase}[1][]{%
    \setkeys{lilyDesignOptions}{scale=1,raise=0}%
    \lilyPrintImage[#1]{quaverDotted}%
}
\newcommand*{\quaverDotted}[1][]{\quaverDottedBase[#1] }
\WithSuffix\newcommand\quaverDotted*[1][]{\quaverDottedBase[#1]}
